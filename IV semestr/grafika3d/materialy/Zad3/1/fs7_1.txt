#version 330 core
out vec4 FragColor;

in vec3 ourColor;
in vec2 TexCoord;

// texture samplers
uniform sampler2D texture1;
uniform sampler2D texture2;
uniform float mixPercentage;

void main()
{
	// mieszanie liniowe dw√≥ch tekstur (80% tekstura 1, 20% tekstura 2)
	FragColor = mix(texture(texture1, TexCoord), texture(texture2, TexCoord), 0.0 + mixPercentage);
}
